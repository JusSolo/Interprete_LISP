( defun fact(n)
    (cond ((= n 1)
        ( 1 ))
    ( * n ( fact  (- n 1 )))))
(print " !5= ")
(fact 5)
(print " !4= ")
(fact 4)


( defun fibo(n)
    (cond ((< n 2)
        (n))
    ( + (fibo (- n 1)) ( fibo  (- n 2 )))))

(print " Primeros valores de la sucesion de Fibonacci: ")
(fibo 1)
(fibo 2)
(fibo 3)
(fibo 4)
(fibo 5)
